--1.Total revenue per day
SELECT
    ORDER_DATE::DATE AS ORDER_DAY,
    SUM(SALES_AMOUNT) AS TOTAL_REVENUE
FROM DW_EXAM_DB.LOCAL.FACT_SALES
GROUP BY ORDER_DAY
ORDER BY ORDER_DAY DESC;

-- 2. Top 5 products by revenue in last 30 days
SELECT
    f.PRODUCT_ID,
    p.PRODUCT_CATEGORY_NAME,
    SUM(f.SALES_AMOUNT) AS TOTAL_REVENUE
    ,MAX(f.ORDER_DATE::DATE) AS MAX_DATE
FROM DW_EXAM_DB.LOCAL.FACT_SALES f
LEFT JOIN DW_EXAM_DB.LOCAL.DIM_PRODUCTS p
    ON f.PRODUCT_ID = p.PRODUCT_ID
WHERE f.ORDER_DATE::DATE BETWEEN (SELECT MAX(ORDER_DATE::DATE) - 30 FROM DW_EXAM_DB.LOCAL.FACT_SALES /* where product_id is not null*/) AND (SELECT MAX(ORDER_DATE::DATE) FROM DW_EXAM_DB.LOCAL.FACT_SALES /*where product_id is not null*/)
/*AND f.product_id is not null*/
GROUP BY f.PRODUCT_ID, p.PRODUCT_CATEGORY_NAME
ORDER BY TOTAL_REVENUE DESC
LIMIT 5;


-- 3. Monthly active customers
SELECT
    DATE_TRUNC('MONTH', f.ORDER_DATE)::DATE AS MONTH,
    COUNT(DISTINCT B.CUSTOMER_UNIQUE_ID) AS ACTIVE_CUSTOMERS
FROM DW_EXAM_DB.LOCAL.FACT_SALES f LEFT JOIN DW_EXAM_DB.LOCAL.DIM_CUSTOMERS b ON F.CUSTOMER_ID = b.CUSTOMER_ID
GROUP BY MONTH
ORDER BY MONTH DESC;

-- 4. Average order value per customer
SELECT
    b.CUSTOMER_UNIQUE_ID,
    COUNT(DISTINCT f.ORDER_ID) AS TOTAL_ORDERS,
    SUM(f.SALES_AMOUNT) AS TOTAL_SPENT,
    SUM(f.SALES_AMOUNT) / COUNT(DISTINCT f.ORDER_ID) AS AVG_ORDER_VALUE
FROM DW_EXAM_DB.LOCAL.FACT_SALES f LEFT JOIN DW_EXAM_DB.LOCAL.DIM_CUSTOMERS b ON F.CUSTOMER_ID = b.CUSTOMER_ID
GROUP BY b.CUSTOMER_UNIQUE_ID
ORDER BY AVG_ORDER_VALUE DESC;


-- 5. Data anomaly detection

-- 5a. Records with price <= 0
SELECT 'PRICE <= 0' AS ANOMALY_TYPE, COUNT(*) AS ROW_COUNT
FROM DW_EXAM_DB.LOCAL.FACT_SALES
WHERE PRICE <= 0 OR PRICE IS NULL;

-- 5b. Records with sales_amount <= 0
SELECT 'SALES_AMOUNT <= 0' AS ANOMALY_TYPE, COUNT(*) AS ROW_COUNT
FROM DW_EXAM_DB.LOCAL.FACT_SALES
WHERE SALES_AMOUNT <= 0 OR SALES_AMOUNT IS NULL;

-- 5c. Records with missing customer_id
SELECT 'MISSING CUSTOMER_ID' AS ANOMALY_TYPE, COUNT(*) AS ROW_COUNT
FROM DW_EXAM_DB.LOCAL.FACT_SALES
WHERE CUSTOMER_ID IS NULL OR CUSTOMER_ID = '';

-- 5d. Records with missing product_id (Causes of data not appearing or not displaying correctly in 2nd Queries (-- 2.))
SELECT 'MISSING PRODUCT_ID' AS ANOMALY_TYPE, ORDER_DATE::DATE, COUNT(*) AS ROW_COUNT,
SUM(COUNT(*)) OVER () AS TOTAL_MISSING
FROM DW_EXAM_DB.LOCAL.FACT_SALES
WHERE PRODUCT_ID IS NULL OR PRODUCT_ID = ''
GROUP BY ORDER_DATE::DATE
ORDER BY ORDER_DATE::DATE DESC;
