-- 1. fact table: fact_sales
CREATE OR REPLACE TABLE DW_EXAM_DB.LOCAL.FACT_SALES AS
WITH A AS (
SELECT
    o.ORDER_ID AS ORDER_ID,
    o.CUSTOMER_ID,
    oi.PRODUCT_ID,
    o.ORDER_PURCHASE_TIMESTAMP AS ORDER_DATE,
    oi.PRICE,
    oi.FREIGHT_VALUE,
    p.PAYMENT_VALUE AS SALES_AMOUNT,
    p.PAYMENT_TYPE,
    ROW_NUMBER()OVER(PARTITION BY o.ORDER_ID ORDER BY o.ORDER_ID) AS SEQ
FROM DW_EXAM_DB.LOCAL.RAW_ORDERS o
LEFT JOIN DW_EXAM_DB.LOCAL.RAW_ORDER_ITEMS oi
    ON o.ORDER_ID = oi.ORDER_ID
LEFT JOIN DW_EXAM_DB.LOCAL.RAW_PAYMENTS p
    ON o.ORDER_ID = p.ORDER_ID
)
SELECT A.* FROM A WHERE A.SEQ = 1;


-- 2. Dimension tables: 
   -- 2a. dim_customers
CREATE OR REPLACE TABLE DW_EXAM_DB.LOCAL.DIM_CUSTOMERS AS
SELECT DISTINCT
    CUSTOMER_ID,
    CUSTOMER_UNIQUE_ID,
    CUSTOMER_ZIP_CODE_PREFIX,
    CUSTOMER_CITY,
    CUSTOMER_STATE
FROM DW_EXAM_DB.LOCAL.RAW_CUSTOMERS;
   
   -- 2b. dim_products
CREATE OR REPLACE TABLE DW_EXAM_DB.LOCAL.DIM_PRODUCTS AS
SELECT DISTINCT
    PRODUCT_ID,
    PRODUCT_CATEGORY_NAME,
    PRODUCT_WEIGHT_G,
    PRODUCT_LENGTH_CM,
    PRODUCT_HEIGHT_CM,
    PRODUCT_WIDTH_CM
FROM DW_EXAM_DB.LOCAL.RAW_PRODUCTS;

   -- 2c. dim_date
CREATE OR REPLACE TABLE DW_EXAM_DB.LOCAL.DIM_DATES AS
SELECT DISTINCT
    YEAR(ORDER_PURCHASE_TIMESTAMP) AS YEAR,
    MONTH(ORDER_PURCHASE_TIMESTAMP) AS MONTH,
    DAY(ORDER_PURCHASE_TIMESTAMP) AS DAY,
    ORDER_PURCHASE_TIMESTAMP::DATE AS FULL_DATE
FROM DW_EXAM_DB.LOCAL.RAW_ORDERS;
